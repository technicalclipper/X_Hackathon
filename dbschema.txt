2. pools
sql
Copy
CREATE TABLE pools (
  id SERIAL PRIMARY KEY,
  pool_type VARCHAR(20) NOT NULL, -- 'TIFO', 'MATCH_VIDEO', 'JERSEY', 'TICKETS'
  match_id TEXT NOT NULL,
  submission_deadline BIGINT NOT NULL,
  voting_deadline BIGINT NOT NULL,
  active BOOLEAN DEFAULT TRUE,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
3. submissions
sql
Copy
CREATE TABLE submissions (
  id SERIAL PRIMARY KEY,
  pool_id INTEGER REFERENCES pools(id) ON DELETE CASCADE,
  creator_address VARCHAR(42) NOT NULL,
  content_url TEXT NOT NULL, -- IPFS/Firebase/etc.
  contract_submission_id INTEGER, -- ID from smart contract
  vote_count INTEGER DEFAULT 0,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
4. votes
sql
Copy
CREATE TABLE votes (
  id SERIAL PRIMARY KEY,
  pool_id INTEGER REFERENCES pools(id) ON DELETE CASCADE,
  submission_id INTEGER REFERENCES submissions(id) ON DELETE CASCADE,
  voter_address VARCHAR(42) NOT NULL,
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
  UNIQUE(pool_id, voter_address) -- one vote per wallet per pool
);

CREATE TABLE auctions (
  token_id INTEGER PRIMARY KEY,
  seller_address VARCHAR(42) NOT NULL,
  min_bid VARCHAR(255) NOT NULL,
  required_psg_tokens VARCHAR(255) NOT NULL,
  highest_bid VARCHAR(255),
  highest_bidder_address VARCHAR(42),
